# remember to set production_url in your _config.yml file!
title : Sitemap
---
<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
    <url>
        <loc>{{ site.production_url }}</loc>
        <lastmod>{{ site.time | date: "%Y-%m-%d" }}</lastmod>
        <changefreq>daily</changefreq>
        <priority>1.0</priority>
    </url>
    {% for post in site.posts %}
    <url>
        <loc>http://{{ site.production_url }}{{ post.url }}</loc>
        {% if post.last_updated %}
        <lastmod>{{ post.last_updated | date: "%Y-%m-%d" }}</lostmod>
        {% elsif post.date %}
        <lastmod>{{ post.date | date: "%Y-%m-%d" }}</lostmod>
        {% else %}
        <lastmod>{{ site.time | date: "%Y-%m-%d" }}</lostmod>
        {% endif %}
        <changefreq>monthly</changefreq>
        <priority>0.9</priority>
    </url>
    {% endfor %}
    {% for page in site.pages %}
    <url>
        <loc>http://{{ site.production_url }}{{ page.url }}</loc>
        {% if page.last_updated %}
        <lastmod>{{ page.last_updated | date: "%Y-%m-%d" }}</lastmod>
        {% elsif page.date %}
        <lastmod>{{ page.date | date: "%Y-%m-%d" }}</lastmod>
        {% else %}
        <lastmod>{{ site.time | date: "%Y-%m-%d" }}</lastmod>
        {% endif %}
        <changefreq>monthly</changefreq>
        <priority>0.8</priority>
    </url>
    {% endfor %}
</urlset>
